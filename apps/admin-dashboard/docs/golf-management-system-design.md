# 골프장 관리 시스템 UI/UX 설계서

## 📌 시스템 개요

### 핵심 개념
- **1코스 = 9홀** (기본 관리 단위)
- **18홀 = 2코스 조합** (전반 9홀 + 후반 9홀)
- **36홀 골프장 = 4개 코스** (A, B, C, D 코스)
- **타임슬롯 = 2개 코스 조합** (18홀 플레이 단위)

### 메뉴 구조
```
├── 🏌️ 골프장 관리 (GolfRounds)
│   ├── 골프장 목록/검색
│   ├── 골프장 상세
│   │   ├── 기본 정보
│   │   ├── 코스 관리 (9홀 단위)
│   │   └── 운영 정보
│   └── 18홀 조합 미리보기
└── 📅 타임슬롯 관리
    ├── 캘린더 뷰
    ├── 목록 뷰
    └── 생성 마법사
```

## 🔄 데이터 구조 및 흐름도

### 데이터 모델
```typescript
// 골프장 (36홀 = 4개 코스)
interface GolfClub {
  id: number;
  name: string;
  totalHoles: number;  // 36
  courses: Course[];   // 4개 코스
}

// 코스 (9홀 단위)
interface Course {
  id: number;
  golfClubId: number;
  name: string;        // "A코스(Lake)"
  code: string;        // "A"
  holes: Hole[9];      // 정확히 9개 홀
  par: number;         // 36 (9홀 합계)
  distance: number;    // 미터 단위
  difficulty: 1-5;
}

// 18홀 조합
interface CourseCombo {
  id: string;          // "A-B"
  frontNine: Course;   // 전반 9홀
  backNine: Course;    // 후반 9홀
  totalPar: number;    // 72 (36+36)
  popularity: number;  // 인기도
  distribution: number; // 분배 비중 (%)
}

// 타임슬롯
interface TimeSlot {
  id: number;
  golfClubId: number;
  comboId: string;     // "A-B"
  date: Date;
  startTime: string;
  maxTeams: number;    // 최대 4팀
  bookedTeams: number;
  price: number;
}
```

### 조합 생성 알고리즘
```typescript
function generateCombos(courses: Course[]): CourseCombo[] {
  const combos: CourseCombo[] = [];
  
  // 36홀 (4코스) 예시: 4×4 = 16가지 (자기자신 포함)
  // 실제로는 12가지 (자기자신 제외 권장)
  
  for (let front of courses) {
    for (let back of courses) {
      if (front.id !== back.id) {  // 다른 코스 조합 권장
        combos.push({
          id: `${front.code}-${back.code}`,
          frontNine: front,
          backNine: back,
          totalPar: front.par + back.par,
          popularity: calculatePopularity(front, back),
          distribution: 0  // 마법사에서 설정
        });
      }
    }
  }
  
  return combos;
}
```
